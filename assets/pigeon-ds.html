Blocos de Código

Integração API
pigeon-pix.js 
// Inicializar cliente PigeonPag
import { PigeonPag } from '@pigeonpag/sdk'

const pigeon = new PigeonPag({
  apiKey: 'pgpag_live_xk92mf...',
  environment: 'production',
})

// Criar cobrança Pix
const charge = await pigeon.pix.create({
  amount: 25000,           // centavos (R$ 250,00)
  description: 'Pedido #1234',
  expiresIn: 3600,         // segundos
  customer: {
    name: 'João Silva',
    cpf:  '123.456.789-00',
  },
})

console.log(charge.qrCode)  // → base64 QR Code
console.log(charge.copyPaste) // → Pix copia e cola
